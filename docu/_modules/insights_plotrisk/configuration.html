[docs]    def get_access_token(self, tenant, username, password):
        conn = http.client.HTTPSConnection("v2sso-dev.cropin.co.in")
        payload = 'client_id=web_app&username=' + username + '&password=' + password + '&grant_type=password'
        headers = {'Content-Type': 'application/x-www-form-urlencoded'}
        conn.request("POST", "/auth/realms/" + tenant + "/protocol/openid-connect/token", payload, headers)
        res = conn.getresponse().read()
        result = json.loads(res.decode())
        return result['access_token']
